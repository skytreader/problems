\chapter{Miscellaneous Problems}

(I.e., problems that were compiled before and left unsolved)

\begin{enumerate}
    \item Let $J(n)$ be the Josephus Numbers\cite{ca-knuth}, and $2^m$ be the
    largest power of 2 not exceeding $n$. Prove that $J(5\cdot 2^m)=2^{m+1}-1$.
    What can you conjecture about $J(x2^m)$ if $x$ is any arbitrary integer?
    \item \textbf{Number Theory?} $2^m-2$ is a multiple of 3 when $m$ is an odd
    but no when $m$ is even.
    \item \textbf{Statistics} A soft drink machine is regulated so that it
    dispenses an average of 200ml per cup. If the amount of drink dispensed is
    normally distributed with a standard deviation equal to 15ml.
    \begin{enumerate}
        \item What fraction of the cups will contain more that 224ml?
        \item What is the probability that a cup contains between 191ml and 209ml?
        \item How many cups will likely overflow if 230ml cups are used for the
        next 1000 drinks?
        \item Below what value do we get the smallest 2.5\% of the drinks?
    \end{enumerate}
\end{enumerate}
